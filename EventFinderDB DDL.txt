CREATE TABLE CAMPUS (CAMPUSID VARCHAR(10) NOT NULL, CAMPUSNAME VARCHAR(100) NOT NULL, CAMPUSSTATE VARCHAR(50) NOT NULL, PRIMARY KEY (CAMPUSID));




CREATE TABLE USERS (USERID VARCHAR(20) NOT NULL, CAMPUSID VARCHAR(10), FULLNAME VARCHAR(100) NOT NULL, EMAIL VARCHAR(100) NOT NULL, PASSWORD VARCHAR(255) NOT NULL, "ROLE" VARCHAR(20) NOT NULL, PHONE VARCHAR(20), FACULTY VARCHAR(100), PRIMARY KEY (USERID));



CREATE TABLE ORGANIZER (ORGANIZERID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1  NOT NULL GENERATED ALWAYS AS IDENTITY, CAMPUSID VARCHAR(10), ORGANIZERNAME VARCHAR(100) NOT NULL, ORGANIZERDETAIL VARCHAR(1000), ORGANIZERTYPE VARCHAR(50), PRIMARY KEY (ORGANIZERID));



CREATE TABLE EVENT (EVENTID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1  NOT NULL GENERATED ALWAYS AS IDENTITY, ORGANIZERID INTEGER, EVENTTITLE VARCHAR(200) NOT NULL, DESCRIPTION VARCHAR(1000), EVENTTYPE VARCHAR(50), EVENTVENUE VARCHAR(200), EVENTDATE DATE, STARTTIME TIME(8), ENDTIME TIME(8), STATUS VARCHAR(20) DEFAULT 'Upcoming' , IMAGEURL VARCHAR(255), MERITPOINTS INTEGER DEFAULT 0 , MAXCAPACITY INTEGER DEFAULT 30 , CAMPUSID VARCHAR(10), PRIMARY KEY (EVENTID));



CREATE TABLE REGISTRATION (REGID INTEGER DEFAULT AUTOINCREMENT: start 1 increment 1  NOT NULL GENERATED ALWAYS AS IDENTITY, USERID VARCHAR(20), EVENTID INTEGER, REGDATE TIMESTAMP DEFAULT CURRENT_TIMESTAMP , STATUS VARCHAR(20) DEFAULT 'Confirmed' , REGTIME TIMESTAMP DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (REGID));
